//SWAPPING of 2 number using 3rd variable
#include<stdio.h>
int main()
{
int a=10, b=20, c;
printf("A=%d B=%d", a,b);
c=a+b;
printf("RESULT=%d", c);
}


//SWAPPING of 2 number without using third variable 

#include<stdio.h>
int main()
{
int a,b,c;
printf("Enter the value of A and B\n");
scanf("%d%d", &a,b);
c=a+b;
printf("RESULT=%d", c);
}
